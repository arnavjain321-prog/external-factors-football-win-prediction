def evaluate_model(name, model, X_train, X_test, y_train, y_test):
    """
    Fit model on training data and compute metrics on test data.
    Returns a dictionary of metrics and the fitted model.
    """
    model.fit(X_train, y_train)

    # Predicted probabilities (for positive class)
    if hasattr(model, "predict_proba"):
        y_proba = model.predict_proba(X_test)[:, 1]
    elif hasattr(model, "decision_function"):
        # Fallback if model has decision_function but not predict_proba
        scores = model.decision_function(X_test)
        # Map scores to 0-1 via a simple min-max (not ideal, but safe fallback)
        s_min, s_max = scores.min(), scores.max()
        y_proba = (scores - s_min) / (s_max - s_min + 1e-9)
    else:
        # As a last resort, use binary predictions
        y_pred = model.predict(X_test)
        y_proba = y_pred.astype(float)

    y_pred = (y_proba >= 0.5).astype(int)

    metrics = {
        "model": name,
        "auc": roc_auc_score(y_test, y_proba),
        "brier": brier_score_loss(y_test, y_proba),
        "log_loss": log_loss(y_test, y_proba),
        "accuracy": accuracy_score(y_test, y_pred),
    }

    cm = confusion_matrix(y_test, y_pred)

    print(f"\n=== {name} ===")
    print("AUC:       ", round(metrics["auc"], 3))
    print("Brier:     ", round(metrics["brier"], 3))
    print("Log loss:  ", round(metrics["log_loss"], 3))
    print("Accuracy:  ", round(metrics["accuracy"], 3))
    print("Confusion matrix (rows: true, cols: predicted):\n", cm)

    return metrics, model, y_proba

models = []

# 1. Logistic Regression (simple baseline)
log_reg = LogisticRegression(
    max_iter=1000,
    solver="lbfgs",
)
models.append(("Logistic Regression", log_reg))

# 2. Random Forest
rf_clf = RandomForestClassifier(
    n_estimators=300,
    max_depth=None,
    min_samples_leaf=5,
    random_state=42,
    n_jobs=-1,
)
models.append(("Random Forest", rf_clf))

# 3. Gradient Boosting (sklearn)
gb_clf = GradientBoostingClassifier(
    random_state=42,
)
models.append(("Gradient Boosting", gb_clf))

# 4. CatBoost (if available)
if CATBOOST_AVAILABLE:
    cat_clf = CatBoostClassifier(
        loss_function="Logloss",
        eval_metric="AUC",
        depth=6,
        learning_rate=0.05,
        l2_leaf_reg=3.0,
        random_seed=42,
        verbose=False,
    )
    models.append(("CatBoost", cat_clf))
else:
    cat_clf = None

    # SHAP can be a bit slow; sampling makes plots faster
    X_sample = X_train.sample(min(2000, len(X_train)), random_state=42)

    explainer = shap.TreeExplainer(tree_model)
    shap_values = explainer.shap_values(X_sample)

    # Bar plot of mean |SHAP| values
